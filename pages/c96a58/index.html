使用C语言对IP地址排序

# 前言

假设有若干IP地址随机存储在文本文件中，需要将其按照从小到大的顺序排序。 程序运行环境：

- windows 10 64-bit
- Visual Studio 2015
1
2

# 解决思路

假设IP地址格式为：a.b.c.d 由观察可知：每个数字的范围在0~255之间，灭个字段都是模256的余数，则可以将所有字段按照所在位置乘以256的对应次幂。 由观察可得公式：(a256^3) + (b256^2) + (c256^1) + (d256^0) 这样我们就可以得到一个长整数。每个IP地址对应一个长整数，只要将IP地址与对应求得的长整数用链表保存起来。这样我们就将问题转换为基数排序，结果的排序即为索求排序，只需用指针或者下标等方法将两者对应。

# 代码实现

// Sort_ip_ddresses.cpp : 定义控制台应用程序的入口点。
//

#include "stdafx.h"
#include <string.h>
#include <stdlib.h>

#define MAX_LINE 128

typedef struct ipAddr {
	char *ip_Str;    // 数据域
	long long ip_Int;
	struct ipAddr *next; // 指针域
}link;

/* 
	创建链表 
*/
link *creatLink()
{
	/* 创建头结点 */
	link * p = (link *)malloc(sizeof(link)); 
	p->next = NULL;
	return p;
}

/*
	往链表增加项
*/
link * addLink(link * head, char *str, long long value)
{
	link *tempLink = head;
	link *targeLink = (link *)malloc(sizeof(link));
	if (targeLink == NULL)
	{
		printf("插入结点前申请内存失败！\n");
		return NULL;
	}
	else
	{
		targeLink->ip_Int = value;
		targeLink->ip_Str = str;
		targeLink->next = NULL;
	}
	while (tempLink->next != NULL)
	{
		tempLink = tempLink->next;
	}
	tempLink->next = targeLink;

	return tempLink;
}


void swapStr(char *str1, char *str2)
{
	char *tempStr;
	tempStr = (char *)malloc(16);
	memset(tempStr, 0, 16);
	strcpy(tempStr, str1);
	strcpy(str1, str2);
	strcpy(str2, tempStr);
	free(tempStr);
}

void swapValue(long long *value1, long long *value2)
{
	long long tempValue = *value1;

	*value1 = *value2;
	*value2 = tempValue;
}
/*
	排序链表【冒泡排序】
*/
link *sortTheLinkedList(link *head)
{
	link *tempLink = NULL;
	bool canSort = true;

	if (head == NULL || head->next == NULL)
		return head;
	while (tempLink != head->next && canSort)
	{
		link *q = head;
		canSort = false;
		q = q->next;
		for (; q->next && q->next != q; q = q->next)
		{
			//if ((strcmp(q->ip_Str, q->next->ip_Str) > 0))  //|| (((strcmp(q->ip_Str, q->next->ip_Str) == 0)) && (strlen(q->ip_Str) > strlen(q->next->ip_Str)))
			if (q->ip_Int > q->next->ip_Int)
			{
				swapStr(q->ip_Str, q->next->ip_Str);
				swapValue(&q->ip_Int, &q->next->ip_Int);
				canSort = true;
			}
		}
		tempLink = q;
	}
	return head;
}


/*
	遍历并输出链表内容
*/
bool showLinkInf(link *head)
{
	int count = 0;
	link *tempLink = head;

	if (tempLink == NULL)
	{
		return false;
	}
	while (tempLink->next != NULL)
	{
		count++;
		tempLink = tempLink->next;
		printf("IP地址：%s，\t大小：%lld\n", tempLink->ip_Str, tempLink->ip_Int);
	}
	printf("链表长度: %d\n\n", count);
	return true;
}
/*
	清空链表
*/
void emptyList(link *head)
{
	link *tempLink;
	
	while (head != NULL)
	{
		tempLink = head;
		head = head->next;
		free(tempLink);
	}
}

/*
	复制字符串
*/
char * strCpy(char *str, int len)
{
	char *tempStr2 = (char *)malloc(sizeof(char)*16);
	memset(tempStr2, 0, 16);
	strncpy(tempStr2, str, len);
	return tempStr2;
}


/* 
	假设IP地址格式为：a.b.c.d
		由观察可知：每个数字的范围在0~255之间，则可以将其看成一个四位的256进制数。
		由观察可得公式：(a*256^3) + (b*256^2) + (c*256^1) + (d*256^0)
*/
long long strToInt(char *str, int n)
{
	char *tokStr, *tempStr;
	long long calcValue = 0;
	long long strTIntVal;

	tempStr = (char *)malloc(sizeof(char)*strlen(str));
	strcpy(tempStr, str);

	tokStr = strtok(tempStr, ".");
	while (tokStr)
	{
		strTIntVal = (long long)atoi(tokStr);
		calcValue = calcValue * 256 + strTIntVal;

		tokStr = strtok(NULL, ".");
	}
	return calcValue;
}


void strTok(link * head, char *str)
{
	char *tokStr, *tempStr;
	long long tempLongData;

	tempStr = (char *)malloc(sizeof(char)*strlen(str));
	strcpy(tempStr, str);

	/* 分割字符串 */
	tokStr = strtok(tempStr, "\n");
	while (tokStr)
	{
		tempLongData = strToInt(tokStr, strlen(tokStr));
		//tempStr = strCpy(tokStr, strlen(tokStr));
		addLink(head, tokStr, tempLongData);			// 加入链表

		tokStr = strtok(NULL, "\n");
	}
}

int main()
{
	char *buf;
	FILE *fp;
	int len, tempLen;
	int linkCount = 0;
	long long tempLongData;
	link * ipLink = creatLink();

	buf = (char *)malloc(sizeof(char)*MAX_LINE);
	fp = fopen("ipAddress.txt", "r");
	if (fp == NULL)
	{
		printf("打开文件失败！\n");
		free(buf);
		free(ipLink);
		return 0;
	}

	while (fgets(buf, MAX_LINE, fp) != NULL)
	{
		len = strlen(buf);
		tempLen = len;
		strTok(ipLink, buf);
	}

	sortTheLinkedList(ipLink);

	if (!showLinkInf(ipLink))	printf("链表为空，遍历失败！\n");

	emptyList(ipLink);
	fclose(fp);
    return 0;
}
